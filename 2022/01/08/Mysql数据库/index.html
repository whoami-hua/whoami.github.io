Mysql数据库

一：Mysql数据库详解

1.登录命令

使用-h参数时连接方式为:TCP/IP

如果没有 -h 参数，默认会使用 UNIX socket 方式

mysql-hhost_name-u user_name–ppassword
2.Mysql程序常用命令
show databases 		//显示所有数据库
use dbname			//选择默认的数据库
show tables    		//显示默认数据库中所有表
\c					//放弃正在输入的命令
\h					//显示命令清单
\q					//退出mysql程序
\s					//查看mysql服务器的状态
3.数据类型种类

4.常用sql语句
1.创建数据库/删除数据库
create database name   			//创建数据库
drop database name 				//删除数据库
2.建立/删除/查看表操作
create table liuyan(id int auto_increment primary key,title varchar(255),content text)
//创建数据表并指定字段
auto_increment             //自增
primary key  				//主键 默认不能为空
drop table name				//删除表
3.更改表结构的操作
alter table name rename as name				//修改liuyan为liuyanban
alter table name add time time				//(first/after 字段名)默认最后
alter table name add primary key (time)		//定义字段为主键
alter table name modify time text				//修改数据类型
alter table name change time user varchar(255)		//修改字段名及数据类型
alter table name drop time							//删除字段

4.插入数据操作
insert into 表名(colname1,colname2) values('value1','value2') //插入单行数据
insert into 表名(title,content) values('title test1','content test1'),('title test1','content test1')				//插入多行数据

5.更新数据
update tbname set cloname='value' where id=1;

6.删除数据
delete from liuyan where id =1;
